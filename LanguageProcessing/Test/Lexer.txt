%usings
%{
using TestProject;
%}

%return {Types}
%errortype {Types.Error}
%endtype {Types.End}

%begin

[int] : { return Types.INT; }
[([1-9])([0-9]*)] : { return Types.Num; }
["((([#-Z^-~])|(\\"|\r|\n|\t|\[|\]))*)"] : { return Types.STRING; }
[([a-zA-Z])(([a-zA-Z0-9])*)] : { return Types.IDENTIFIER; }
[\w(\w*)] : { return Lex(); }

%end
